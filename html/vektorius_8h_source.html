242 iterator insert(const_iterator pos, const T& value) {
243 size_type index = pos - begin();
244 if (mdydis_ == mtalpa_) {
245 size_type new_capacity = (mtalpa_ == 0) ? 1 : mtalpa_ * 2;
246 reserve(new_capacity);

249 // Perstumiam visus elementus nuo įterpimo vietos į dešinę per viena

274 iterator erase(const_iterator first, const_iterator last)
275 {
276 size_type first_index = first - begin();
277 size_type last_index = last - begin();
278 if (first_index > last_index || last_index > mdydis_) {
279 throw std::out_of_range("Invalid range");
280 }
281
282 size_type num_to_erase = last_index - first_index;
283 for (size_type i = first_index; i < mdydis_ - num_to_erase; ++i) {
284 mduom_[i] = std::move(mduom_[i + num_to_erase]);
285 }
286
287 mdydis_ -= num_to_erase;
288 return begin() + first_index;
